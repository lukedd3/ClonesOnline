28  public class SettingsManager {
29      private static final String SETTINGS_PREFERENCES_NAME = "SETTINGS";
30      
31      public static boolean getSettingValue(SettingType setting){
32          Preferences prefs = Gdx.app.getPreferences(SETTINGS_PREFERENCES_NAME);
33          boolean settingValue = prefs.getBoolean(setting.toString(), setting.getDefaultValue());
34          return settingValue;
35      }
36      
37      public static void setSettingValue(SettingType setting, boolean newValue){
38          Preferences prefs = Gdx.app.getPreferences(SETTINGS_PREFERENCES_NAME);
39          prefs.putBoolean(setting.toString(), newValue);
40          prefs.flush();
41      }
42      
43  }
