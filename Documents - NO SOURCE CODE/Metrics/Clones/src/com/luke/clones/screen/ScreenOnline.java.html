449     private synchronized void sortRoomInfoList(RoomInfo[] roomInfoList, final int sortColumn){
450         if(roomInfoList==null) return;
451         Arrays.sort(roomInfoList, new Comparator<RoomInfo>() {
452             @Override
453             public int compare(RoomInfo arg0, RoomInfo arg1) {
454                 switch(sortColumn){
455                 case 0:
456                     int freePlaces0 = arg0.getPlayerLimit()-arg0.getNumberOfPlayers();
457                     int freePlaces1 = arg1.getPlayerLimit()-arg1.getNumberOfPlayers();
458                     if(freePlaces0<freePlaces1) return 1;
459                     else if(freePlaces0>freePlaces1) return -1;
460                     else return 0;
461                 case 1:
462                     return arg0.getRoomName().compareTo(arg1.getRoomName());
463                 case 2:
464                     return arg0.getAdminName().compareTo(arg1.getAdminName());
465                 default:
466                     return 0;
467                 }
468                 
469             }
470         });
471     }

1257    @Override
1258    public synchronized void received(Connection connection, Object obj) {
1259        if(obj instanceof UpdateRoom){
1260            NetworkConfig.log("received update room");
1261            final UpdateRoom updateRoom = (UpdateRoom) obj;
1262            ActionTypeUpdateRoom actionType = updateRoom.getActionType();
1263            if(actionType==ActionTypeUpdateRoom.ROOM_LIST && creatingRoom==false){
1264                roomInfoList = updateRoom.getRoomInfoList();
1265                
1266                buildRoomList(roomInfoList, sortColumn);
1267                
1268                NetworkConfig.log("Room Info List:");
1269                if(roomInfoList!=null){
1270                    for(RoomInfo roomInfo: roomInfoList){
1271                        String playerString = "";
1272                        for(String playerName: roomInfo.getPlayerNames()){
1273                            playerString += playerName + ", ";
1274                        }
1275                        NetworkConfig.log(roomInfo.getRoomId()+" | "+roomInfo.getRoomName()+" | "+playerString);
1276                    }
1277                }
1278            }
1279            else if(actionType==ActionTypeUpdateRoom.FORBIDDEN_NAME){
1280                NetworkConfig.log("Room name forbidden");
1281                TextButton tb = new TextButton("ok", skin);
1282                errorDialogNameTooShort = new CustomDialog(smallSkin);
1283                Label errorLabel = new Label("This room name is too short", skin);
1284                errorLabel.setColor(Color.RED);
1285                errorDialogNameTooShort.add(errorLabel);
1286                errorDialogNameTooShort.row();
1287                tb.addListener(new ClickListener(){
1288                    @Override
1289                    public void clicked(InputEvent event, float x, float y) {
1290                        super.clicked(event, x, y);
1291                        SoundManager.INSTANCE.play(SoundType.CLICK);
1292                        errorDialogNameTooShort.remove();
1293                    }
1294                });
1295                errorDialogNameTooShort.add(tb);
1296                mainStage.addActor(errorDialogNameTooShort);
1297            }
1298            else if(actionType==ActionTypeUpdateRoom.NAME_ALREADY_TAKEN){
1299                NetworkConfig.log("Room name already taken");
1300                TextButton tb = new TextButton("ok", skin);
1301                errorDialogNameAlreadyTaken = new CustomDialog(smallSkin);
1302                Label errorLabel = new Label("This room name is already taken", skin);
1303                errorLabel.setColor(Color.RED);
1304                errorDialogNameAlreadyTaken.add(errorLabel);
1305                errorDialogNameAlreadyTaken.row();
1306                tb.addListener(new ClickListener(){
1307                    @Override
1308                    public void clicked(InputEvent event, float x, float y) {
1309                        super.clicked(event, x, y);
1310                        SoundManager.INSTANCE.play(SoundType.CLICK);
1311                        errorDialogNameAlreadyTaken.remove();
1312                    }
1313                });
1314                errorDialogNameAlreadyTaken.add(tb);
1315                mainStage.addActor(errorDialogNameAlreadyTaken);
1316            }
1317            else if(actionType==ActionTypeUpdateRoom.CREATE_SUCCESS){
1318                buildRoomInterior(updateRoom.getRoomInterior());
1319                NetworkConfig.log("Room create success");
1320                if(errorDialogNameTooShort!=null) errorDialogNameTooShort.remove();
1321                if(errorDialogNameAlreadyTaken!=null) errorDialogNameAlreadyTaken.remove();
1322            }
1323            else if(actionType==ActionTypeUpdateRoom.ENTER_SUCCESS){
1324                buildRoomInterior(updateRoom.getRoomInterior());
1325                NetworkConfig.log("Room enter success");
1326            }
1327            else if(actionType==ActionTypeUpdateRoom.ROOM_INTERIOR){
1328                buildRoomInterior(updateRoom.getRoomInterior());
1329                NetworkConfig.log("Room interior updated");
1330            }
1331            if(actionType==ActionTypeUpdateRoom.BACK_TO_ROOM_LIST){
1332                buildRoomListUpperGUI();
1333                
1334                Gdx.app.postRunnable(new Runnable() {
1335                    public void run() {
1336                        List<QuickMessage> quickMessagesAboutAchievements = QuickMessageUtil.prepareQuickMessagesWithNewAchievements(updateRoom.getNewAchievements(), skin, smallSkin, startScreen);
1337                        for(QuickMessage quickMessage : quickMessagesAboutAchievements){
1338                            QuickMessageManager.showMessageOnSpecifiedStage(quickMessage, mainStage);
1339                        }
1340                    }
1341                });
1342                
1343                interstitialAdsManager.afterLeaveRoom(startScreen.getAndroidNativeHandler(), mainStage, skin);
1344                 
1345                clonesClient.send(new MoveRoom(ActionTypeMoveRoom.GET_LIST, null, null));
1346            }
1347            if(actionType==ActionTypeUpdateRoom.BACK_TO_ROOM_INTERIOR){
1348                clonesClient.send(new MoveRoom(ActionTypeMoveRoom.GET_INTERIOR, null, null));
1349            }
1350            if(actionType==ActionTypeUpdateRoom.MODIFY_REQUEST_RESPONSE){
1351                buildRoomModify(updateRoom.getRoomInterior());
1352            }
1353        }
1354        else if(obj instanceof GameStart){
1355            final GameStart gameStart = (GameStart) obj;
1356             Gdx.app.postRunnable(new Runnable() {
1357                 public void run() {
1358                     startScreen.setToOnlineGame(clonesClient, gameStart, playerName);
1359                 }});
1360             startScreen.getAndroidNativeHandler().hideBannerAds();
1361        }
1362        else if(obj instanceof StatsResponse){
1363            StatsResponse statsResponse = (StatsResponse) obj;
1364            ActionTypeStats actionTypeStats = statsResponse.getActionTypeStats();
1365            
1366            if(actionTypeStats==ActionTypeStats.GET_GENERAL_STATS){
1367                buildStatistics(statsResponse.getLevel(), statsResponse.getPercentToNextLevel(),
1368                        ClientStatsUtil.statsValuesToPercent(statsResponse.getStatsMap()),statsResponse.getStatsMap());
1369            }
1370        }
1371        else if(obj instanceof AchievementsResponse){
1372            AchievementsResponse achievementsResponse = (AchievementsResponse) obj;
1373            buildAchievements(achievementsResponse.getCurrentAchievements());
1374        }
1375    }
