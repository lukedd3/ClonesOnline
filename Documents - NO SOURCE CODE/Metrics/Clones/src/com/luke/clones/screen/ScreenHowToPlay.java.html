64      public ScreenHowToPlay(StartScreen startScreen){

68      @Override
69      public void show() {
70          width = Config.width;
71          height = Config.height;
72  //      stage = new Stage(width, height, false);
73          viewport = new FitViewport(width, height);
74          stage = new Stage();
75          stage.setViewport(viewport);
76          
77          specialKeyHandler = new SpecialKeyHandler() {
78              @Override
79              public void backKeyAction() {
80                  startScreen.setToMainMenu();
81              }
82          };
83          inputMultiplexer = new InputMultiplexer(stage, specialKeyHandler);
84          
85          Gdx.input.setInputProcessor(inputMultiplexer);
86          skin = new Skin(Gdx.files.internal("data/skins/skin_medium.json")); 
87          
88          buildContent();
89      }
90      
91      private synchronized void buildContent(){
92          prepareMainTable();
93          prepareContent();
94      }
95      
96      private void prepareMainTable() {
97          mainTable = new Table(skin);
98          mainPane = new ScrollPane(mainTable, skin);
99          mainPane.setPosition(0+Config.widthOffset/2, Config.menuBottomHeightOffset);
100         mainPane.setSize(width-Config.widthOffset/2, height-(Config.menuBottomHeightOffset));
101         mainPane.setScrollingDisabled(true, false);
102         mainPane.setupFadeScrollBars(0.5f, 0);
103         
104 //      mainTable.setPosition(0+Config.widthOffset/2, Config.menuBottomHeightOffset);
105         mainTable.setSize(width-Config.widthOffset/2, height-(Config.menuBottomHeightOffset+Config.menuTopHeightOffset));
106         mainTable.setBackground(skin.getDrawable("default-round-down"));
107         mainTable.align(Align.top);
108         
109         stage.addActor(mainPane);
110     }
