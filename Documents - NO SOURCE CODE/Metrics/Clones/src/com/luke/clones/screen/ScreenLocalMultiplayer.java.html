59      private StartScreen startScreen;

69      public ScreenLocalMultiplayer(StartScreen startScreen){

73      @Override
74      public void show() {
75          width = Config.width;
76          height = Config.height;
77  //      stage = new Stage(width, height, false);
78          viewport = new FitViewport(width, height);
79          stage = new Stage();
80          stage.setViewport(viewport);
81          
82          specialKeyHandler = new SpecialKeyHandler() {
83              @Override
84              public void backKeyAction() {
85                  startScreen.setToMainMenu();
86              }
87          };
88          inputMultiplexer = new InputMultiplexer(stage, specialKeyHandler);
89          Gdx.input.setInputProcessor(inputMultiplexer);
90          
91          skin = new Skin(Gdx.files.internal("data/skins/skin_medium.json"));
92          
93          buildGameOptions();
94      }
95  
96      @Override
97      public void hide() {
98          stage = null;
99          skin = null;
100     }

112         Table mainTable = new Table(skin);

120         mainTable.align(Align.top);

124         //Back button
125         TextButton backButton = new TextButton("Back to menu", skin);

129             @Override
130             public void clicked(InputEvent event, float x, float y) {
131                  Gdx.app.postRunnable(new Runnable() {
132                      public void run() {
133                          SoundManager.INSTANCE.play(SoundType.CLICK);

135                      }});
136             }
137         });
138         
139         //Players Number Label
140         TextButton playersNumLabel = new TextButton("Players Number", skin);
141         playersNumLabel.setDisabled(true);
142         mainTable.add(playersNumLabel).minSize(width-6, 70).spaceBottom(3).spaceTop(3);

146

185         final String[] fourPlayerMapNames = MapModelUtils.resolveMapsToNames(fourPlayerMaps);
186

220         startGameButton.addListener(new ClickListener(){

223                 try {
224 //                  String mapName = mapSelectBox.getSelection();
225                     SoundManager.INSTANCE.play(SoundType.CLICK);
226                     String mapName = mapSelectBox.getSelected();
227                     Map map = MapAutoCreator.createMap(MapModelUtils.resolveNameToMap(mapName));
228 //                  PlayerMoveTimeLimitType playerMoveTimeLimitType = playerMoveTimeLimitSelectBoxItems[playerMoveTimeLimitSelectBox.getSelectionIndex()];
229                     PlayerMoveTimeLimitType playerMoveTimeLimitType = playerMoveTimeLimitSelectBoxItems[playerMoveTimeLimitSelectBox.getSelectedIndex()];

231                     startScreen.getAndroidNativeHandler().hideBannerAds();
232                 } catch (MapCreatorArgumentException e) {

234                 }
235             }
236         });
237         mainTable.add(startGameButton).minSize(width-6, 70).spaceBottom(3).spaceTop(3);
238     }

240     private MapModel[] selectMapsWithSpecifiedNumberOfPlayers(int numberOfPlayers){
241         ArrayList<MapModel> selectedMaps = new ArrayList<MapModel>();
242         for(AvailableMapModelsType availableMapModelsType: AvailableMapModelsType.values()){
243             if(availableMapModelsType.getMap().getNumberOfPlayers()==numberOfPlayers){
244                 selectedMaps.add(availableMapModelsType.getMap());
245             }
246         }
247         return selectedMaps.toArray(new MapModel[0]);
248     }
