200     @Override
201     public synchronized void received(Connection connection, Object obj) {
202         //if login ok -> goto screenOnline
203         //if login not ok -> try again, give info what is the problem
204         if(obj instanceof UpdateLogin){
205             UpdateLogin updateLogin = (UpdateLogin) obj;
206             ActionTypeUpdateLogin action = updateLogin.getActionType();
207             if(action == ActionTypeUpdateLogin.SUCCESS){
208                 errorLabel.setText("");
209                  Gdx.app.postRunnable(new Runnable() {
210                      public void run() {
211                          startScreen.setToOnline(clonesClient, playerName);
212                      }});
213                 System.out.println("Udane logowanie !!!");
214             }
215             else if(action == ActionTypeUpdateLogin.INVALID_AUTH_TOKEN){
216                 authenticationFailureBehaviour();
217             }
218             else if(action == ActionTypeUpdateLogin.SERVER_FULL){
219                 serverFullBehaviour();
220             }
221             else if(action == ActionTypeUpdateLogin.UNEXPECTED_ERROR){
222                 unexpectedErrorBehaviour();
223             }
224         }
225     }

306     private void login(){
307         boolean isConnectedToServer = connectToServer();
308         
309         androidPickAccountHandler=new AndroidPickAccountHandler() {
310             @Override
311             public void onCancelOrError() {
312                 System.out.println("onCancelOrError");
313                 cancelBehaviour();
314             }
315             @Override
316             public void onAccountPicked(String accountName) {
317                 //idzie têdy
318                 System.out.println("onAccountPicked");
319                 startScreen.getAndroidNativeHandler().getGoogleAccountToken(accountName, androidAuthenticationTokenHandler);
320             }
321         };
322         
323         androidAuthenticationTokenHandler=new AndroidAuthenticationTokenHandler() {
324             @Override
325             public void onSuccess(String token, String username) {
326                 System.out.println("onSuccess");
327 //              System.out.println(token);
328                 playerName = NameUtil.getNameFromEmail(username);
329                 clonesClient.send(new MoveLogin(token, playerName));
330             }
331             @Override
332             public void onIOException(Exception e) {
333                 System.out.println("onIOException");
334                 authenticationServerIOExceptionBehaviour();
335                 e.printStackTrace();
336             }
337             @Override
338             public void onAuthException(Exception e) {
339                 //idzie têdy
340                 System.out.println("onAuthException");
341                 authenticationFailureBehaviour();
342             }
343         };
344         
345         if(isConnectedToServer){
346             startScreen.getAndroidNativeHandler().pickGoogleAccount(androidPickAccountHandler);
347         }
348         
349     }

351     private boolean connectToServer(){
352         clonesClient = new ClonesClient();
353         try {
354             int port = serverInfo.getPort();
355             String host = serverInfo.getHost();
356             clonesClient.start(this, host, port);
357             return true;
358         } catch (IOException e) {
359             System.out.println("Klient nie wystartowa³");
360             serverIOExceptionBehaviour();
361             e.printStackTrace();
362             return false;
363         }
364     }
