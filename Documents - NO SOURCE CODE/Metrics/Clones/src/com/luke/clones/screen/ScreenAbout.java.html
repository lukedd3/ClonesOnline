64      @Override
65      public synchronized void show() {
66          width = Config.width;
67          height = Config.height;
68  //      stage = new Stage(width, height, false);
69          viewport = new FitViewport(width, height);
70          stage = new Stage();
71          stage.setViewport(viewport);
72          
73          specialKeyHandler = new SpecialKeyHandler() {
74              @Override
75              public void backKeyAction() {
76                  startScreen.setToMainMenu();
77              }
78          };
79          inputMultiplexer = new InputMultiplexer(stage, specialKeyHandler);
80          
81          Gdx.input.setInputProcessor(inputMultiplexer);
82          skin = new Skin(Gdx.files.internal("data/skins/skin_medium.json")); 
83          smallSkin = new Skin(Gdx.files.internal("data/skins/skin_small.json")); 
84          
85          buildContent();
86      }
