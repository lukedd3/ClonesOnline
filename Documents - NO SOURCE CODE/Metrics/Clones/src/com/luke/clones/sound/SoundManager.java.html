33  public class SoundManager {
34      public static final SoundManager INSTANCE = new SoundManager();
35      
36      Map<SoundType, Sound> soundMap;
37      
38      private SoundManager(){
39          soundMap = new HashMap<SoundType, Sound>();
40          soundMap.put(SoundType.JUMP, Gdx.audio.newSound(Gdx.files.internal("data/sounds/jump.wav")));
41          soundMap.put(SoundType.CLONE, Gdx.audio.newSound(Gdx.files.internal("data/sounds/clone.wav")));
42          soundMap.put(SoundType.LOSE, Gdx.audio.newSound(Gdx.files.internal("data/sounds/lose.wav")));
43          soundMap.put(SoundType.WIN, Gdx.audio.newSound(Gdx.files.internal("data/sounds/win.wav")));
44          soundMap.put(SoundType.CLICK, Gdx.audio.newSound(Gdx.files.internal("data/sounds/click2.wav")));
45      }
46      
47      public void play(SoundType soundToPlayType){
48          if(isSoundOn()){
49              playSoundWithFullVolume(soundToPlayType);
50          }
51      }
52      
53      public void play(SoundType soundToPlayType, float volume){
54          if(isSoundOn()){
55              playSoundWithSpecifiedVolume(soundToPlayType, volume);
56          }
57      }
58      
59      private boolean isSoundOn(){
60          return SettingsManager.getSettingValue(SettingType.SOUND_ON);
61      }
62      
63      private void playSoundWithFullVolume(SoundType soundToPlayType){
64          Sound soundToPlay = soundMap.get(soundToPlayType);
65          soundToPlay.play(soundToPlayType.getVolume());
66      }
67      
68      private void playSoundWithSpecifiedVolume(SoundType soundToPlayType, float volume){
69          Sound soundToPlay = soundMap.get(soundToPlayType);
70          soundToPlay.play(volume);
71      }
72      
73  }
