29  public class Map {
30      private int width; // wide side
31      private int height; // narrow side
32      private Position[] startSolids; //obstacle fields
33      private PlayerPosition[] startPlayers; // where player clones will be at the start
34      private MapType mapType;
35      
36      /**
37       *For serialization
38       */
39      public Map(){
40          
41      }
42      
43      public Map(int width, int height, Position[] startSolids, PlayerPosition[] startPlayers, MapType mapType){
44          this.width = width;
45          this.height = height;
46          this.startSolids = startSolids;
47          this.startPlayers = startPlayers;
48          this.mapType = mapType;
49      }
50      
51      public int getWidth() {
52          return width;
53      }
54      
55      public int getHeight() {
56          return height;
57      }
58      
59      public Position[] getStartSolids() {
60          return startSolids;
61      }
62      
63      public PlayerPosition[] getStartPlayers() {
64          return startPlayers;
65      }
66  
67      public MapType getMapType() {
68          return mapType;
69      }
70  
71      public void setStartPlayers(PlayerPosition[] startPlayers) {
72          this.startPlayers = startPlayers;
73      }
74      
75  }
