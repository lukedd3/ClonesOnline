33  public class FieldActor2 extends Actor{
34      private Texture emptyTexture, solidTexture, ballTexture, backgroundTexture;
35      private Color fieldColor;
36      private Position position;
37      private FieldType type;
38      
39      public FieldActor2(Texture emptyTexture, Texture solidTexture, Texture ballTexture, Texture backgroundTexture, Color color, Position position, FieldType fieldType){
40          this.emptyTexture = emptyTexture;
41          this.solidTexture = solidTexture;
42          this.ballTexture = ballTexture;
43          this.backgroundTexture = backgroundTexture;
44          type = fieldType;
45          if(color!=null) fieldColor = color;
46          this.position = position;
47      }
48  
49  //  @Override
50  //  public void draw(SpriteBatch sb, float parentAlpha) {
51      @Override
52      public void draw(Batch sb, float parentAlpha) {
53          if(type == FieldType.EMPTY){
54              if(fieldColor!=null) sb.setColor(fieldColor);
55              else sb.setColor(Color.WHITE);
56              sb.draw(emptyTexture, getX(), getY(), getWidth(), getHeight());
57          }
58          else if(type == FieldType.SOLID){
59              if(fieldColor!=null) sb.setColor(fieldColor);
60              else sb.setColor(Color.WHITE);
61              sb.draw(solidTexture, getX(), getY(), getWidth(), getHeight());
62          }
63          else if(type == FieldType.TAKEN){
64              sb.setColor(Color.WHITE);
65              sb.draw(backgroundTexture, getX(), getY(), getWidth(), getHeight());
66              if(fieldColor!=null) sb.setColor(fieldColor);
67              sb.draw(ballTexture, getX(), getY(), getWidth(), getHeight());
68          }
69      }
70      
71      public Position getPosition(){
72          return position;
73      }
74      
75      public void setFieldType(FieldType fieldType){
76          type = fieldType;
77      }
78      
79      public FieldType getFieldType(){
80          return type;
81      }
82      
83      public void setColor(Color color){
84          //if(color!=null)super.setColor(color);
85          fieldColor = color;
86      }
87      
88      @Override
89      public void setColor(float r, float g, float b, float a) {
90          fieldColor.set(r, g, b, a);
91      }
92  
93      @Override
94      public Color getColor() {
95          return fieldColor;
96      }
97      
98  }
