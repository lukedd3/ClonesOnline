33  public class HighlightActor extends Actor{
34      private TextureRegion fieldTextureRegion;
35      protected Color fieldColor;
36      private Position position;
37      
38      public HighlightActor(TextureRegion fieldTextureRegion, Color color, Position position){
39          this.fieldTextureRegion = fieldTextureRegion;
40          if(color!=null) fieldColor = color;
41          this.position = position;
42      }
43      
44      public HighlightActor(Texture fieldTexture, Color color, Position position){
45          this(new TextureRegion(fieldTexture), color, position);
46      }
47      
48  //  @Override
49  //  public void draw(SpriteBatch sb, float parentAlpha) {
50      @Override
51      public void draw(Batch sb, float parentAlpha) {
52          if(fieldColor!=null) sb.setColor(fieldColor);
53          else sb.setColor(Color.WHITE);
54          sb.draw(fieldTextureRegion, getX(), getY(), getWidth(), getHeight());
55      }
56      
57      public Position getPosition(){
58          return position;
59      }
60      
61  }
