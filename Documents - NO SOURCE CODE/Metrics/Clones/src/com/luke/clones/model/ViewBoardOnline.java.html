422                     if(playerType == PlayerType.BLUE) color = Color.BLUE;
423                     else if(playerType == PlayerType.GREEN) color = Color.GREEN;

426                     else color = Color.WHITE;

428                     ballField.setPosition((j*fieldSize)+unusedSpace/2, i*fieldSize);

430                     ballField.addListener(fieldListener);
431                     mainStage.addActor(ballField);

439         startCountingPlayerMoveTime();
440

451         statusBar.reflesh();
452         for(int i=0; i<board.getWidth();i++){
453             for(int j=0; j<board.getHeight();j++){
454                 FieldType fieldType = board.getFieldType(i, j);
455                 Position pos = new Position(i,j);
456                 FieldActor2 field = fieldMap.get(pos);
457                     if(fieldType == FieldType.TAKEN){
458                         PlayerType playerType = board.getFieldPlayer(i, j);
459                         Color color;
460                         if(playerType == PlayerType.BLUE) color = Color.BLUE;
461                         else if(playerType == PlayerType.GREEN) color = Color.GREEN;
462                         else if(playerType == PlayerType.RED) color = Color.RED;
463                         else if(playerType == PlayerType.ORANGE) color = Color.ORANGE;
464                         else color = Color.WHITE;
465                         if(field.getColor() != color) field.setColor(color);
466                         if(field.getFieldType() != fieldType) field.setFieldType(fieldType);
467                     }
468                     else{
469                         if(field.getFieldType() != fieldType) {
470                             field.setFieldType(fieldType);
471                             field.setColor(null);
472                         }
473                     }
474             }
475         }
476         
477         flashCurrentPlayerBalls();
478         
479         startCountingPlayerMoveTime();
480         
481         if(Config.debug) System.out.println(" Red: "+board.getBallCount(PlayerType.RED)+" Blue: "+board.getBallCount(PlayerType.BLUE)+" Green: "+board.getBallCount(PlayerType.GREEN)+" Orange: "+board.getBallCount(PlayerType.ORANGE));
482     }

496     private class FieldListener extends ClickListener{
497 
498         @Override
499         public synchronized void clicked(InputEvent event, float x, float y) {
500             super.clicked(event, x, y);
501             FieldActor2 field = (FieldActor2) event.getListenerActor();
502             Position pos = field.getPosition();
503             PlayerType fieldPlayer = board.getFieldPlayer(pos.getX(), pos.getY());
504             if(board.getFieldType(pos.getX(), pos.getY()) == FieldType.TAKEN && fieldPlayer == turn.getCurrentPlayer()){
505                 clickedPlayerPosition = new PlayerPosition(pos.getX(), pos.getY(), fieldPlayer);
506                 
507                 highlightPossibleMoveFields(clickedPlayerPosition);
508             }
509             else if(board.getFieldType(pos.getX(), pos.getY()) == FieldType.EMPTY){
510                 if(clickedPlayerPosition!=null){
511                     if(board.isMovePossible(clickedPlayerPosition.getX(), clickedPlayerPosition.getY(), pos.getX(), pos.getY())){
512                         Config.log("My player Type: "+playerColor);
513                         clonesClient.send(new MoveBoard(ActionTypeMove.MOVE,clickedPlayerPosition,pos));
514                         clickedPlayerPosition=null;
515                         
516                         layerStage.clear();
517                     }
518                 }
519             }
520         }
521 
522     }

524     private void highlightPossibleMoveFields(PlayerPosition chosenPlayerPosition){
525         if(chosenPlayerPosition.getPlayerType() == playerColor){
526             List<Position> possibleCloneFieldsList = new ArrayList<Position>();
527             List<Position> possibleJumpFieldsList = new ArrayList<Position>();
528             
529             for(int i= chosenPlayerPosition.getX()-2; i<=chosenPlayerPosition.getX()+2;i++){
530                 for(int j= chosenPlayerPosition.getY()-2;j<=chosenPlayerPosition.getY()+2;j++){
531                     if(board.isClonePossible(chosenPlayerPosition.getX(), chosenPlayerPosition.getY(), i, j)){
532                         possibleCloneFieldsList.add(new Position(i, j));
533                     }
534                     if(board.isJumpPossible(chosenPlayerPosition.getX(), chosenPlayerPosition.getY(), i, j)){
535                         possibleJumpFieldsList.add(new Position(i, j));
536                     }
537                 }
538                 
539                 layerStage.clear();
540                 
541                 PlayerType playerType = chosenPlayerPosition.getPlayerType();
542                 Color playerColor = convertPlayerTypeToPlayerColor(playerType);
543                 
544                 Color markBallColor = new Color(playerColor.r, playerColor.g, playerColor.b, 0.4f);

601     private List<Position> getCurrentPlayerBallsPositions(){
602         List<Position> balls = new ArrayList<Position>();
603 
604         for(int i=0; i<board.getWidth(); i++){
605             for(int j=0; j<board.getHeight(); j++){
606                 boolean fieldWithBallWithPlayerColor = board.getFieldType(i, j)==FieldType.TAKEN && 
607                         board.getFieldPlayer(i, j)==playerColor;
608                 if(fieldWithBallWithPlayerColor){
609                     balls.add(new Position(i,j));
610                 }
611             }
612         }
613         
614         return balls;
615     }

618         Color playerColor;
619         if(playerType == PlayerType.BLUE) playerColor = Color.BLUE;
620         else if(playerType == PlayerType.GREEN) playerColor = Color.GREEN;
621         else if(playerType == PlayerType.RED) playerColor = Color.RED;
622         else if(playerType == PlayerType.ORANGE) playerColor = Color.ORANGE;
