93      private void login(final TimeMeasure time){
94          boolean isConnectedToServer = connectToServer();
95          
96          androidPickAccountHandler=new AndroidPickAccountHandler() {
97              @Override
98              public void onCancelOrError() {
99                  System.out.println("onCancelOrError");
100                 cancelBehaviour();
101             }
102             @Override
103             public void onAccountPicked(String accountName) {
104                 System.out.println("onAccountPicked");
105                 androidNativeHandler.getGoogleAccountToken(accountName, androidAuthenticationTokenHandler);
106             }
107         };
108         
109         androidAuthenticationTokenHandler=new AndroidAuthenticationTokenHandler() {
110             @Override
111             public void onSuccess(String token, String username) {
112                 System.out.println("onSuccess");
113                 playerName = NameUtil.getNameFromEmail(username);
114                 time.start();
115                 clonesClient.send(new MoveLogin(token, playerName));
116             }
117             @Override
118             public void onIOException(Exception e) {
119                 System.out.println("onIOException");
120                 authenticationServerIOExceptionBehaviour();
121                 e.printStackTrace();
122             }
123             @Override
124             public void onAuthException(Exception e) {
125                 System.out.println("onAuthException");
126                 authenticationFailureBehaviour();
127             }
128         };
129         
130         if(isConnectedToServer){
131             androidNativeHandler.pickGoogleAccount(androidPickAccountHandler);
132         }
133         else{
134             connectTried = true;
135             connected = false;
136         }
137

140     private boolean connectToServer(){
141         clonesClient = new ClonesClient();
142         try {
143             int port = serverInfo.getPort();
144             String host = serverInfo.getHost();
145             clonesClient.start(this, host, port);
146             return true;
147         } catch (IOException e) {
148             System.out.println("Klient nie wystartowa³");
149             serverIOExceptionBehaviour();
150             e.printStackTrace();
151             return false;
152         }
153     }

159     @Override
160     public synchronized void received(Connection connection, Object obj) {
161         if(obj instanceof UpdateLogin){
162             UpdateLogin updateLogin = (UpdateLogin) obj;
163             ActionTypeUpdateLogin action = updateLogin.getActionType();
164             if(action == ActionTypeUpdateLogin.SUCCESS){
165                 connected = true;
166                 System.out.println("Udane logowanie !!!");
167             }
168             else if(action == ActionTypeUpdateLogin.INVALID_AUTH_TOKEN){
169                 authenticationFailureBehaviour();
170                 connected = false;
171             }
172             else if(action == ActionTypeUpdateLogin.SERVER_FULL){
173                 serverFullBehaviour();
174                 connected = false;
175             }
176             else if(action == ActionTypeUpdateLogin.UNEXPECTED_ERROR){
177                 unexpectedErrorBehaviour();
178                 connected = false;
179             }
180             connectTried = true;
181         }
182     }
